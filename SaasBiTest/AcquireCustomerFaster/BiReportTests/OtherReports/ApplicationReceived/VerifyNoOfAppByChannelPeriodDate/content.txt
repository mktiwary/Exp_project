!|Store Query|!- /*Temp query to get required data*/
with app_rec as(
select u.applicationnumber,applicationcreationlocaltimestamp, u.applicationupdatelocaltimestamp, 
u.decisiontype,u.applicationstatus,u.channel,u.worklist, systemdecision,applicationsource
from public.f_application_decision as u
inner join (
select applicationNumber,producttype,productid, max(applicationupdatelocaltimestamp) applicationupdatelocaltimestamp from public.f_application_decision
group by applicationNumber,productid,producttype
order by applicationnumber
) as q
on u.applicationnumber =q.applicationnumber
and u.applicationupdatelocaltimestamp = q.applicationupdatelocaltimestamp
 
order by u.applicationupdatelocaltimestamp
) 

select to_char (applicationcreationlocaltimestamp, 'YYYY-MM-DD')::date date1
, count(distinct applicationnumber) no_of_app, channel
from app_rec
--where to_char (applicationcreationlocaltimestamp, 'YYYY-MM-DD')='2017-01-01'
group by to_char (applicationcreationlocaltimestamp, 'YYYY-MM-DD'), channel
order by to_char (applicationcreationlocaltimestamp, 'YYYY-MM-DD'), channel
 -!|source_data|
