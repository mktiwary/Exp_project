!|Store Query|!- select decisioner, sum(sd+md)::int as Count from
(
select decisioner,
sum(case when decision_type ='System' then 1 else 0 end) as SD,

)x group by decisioner

!* Compare source data with target data
!*< Setup

!*> Source Query
!|Store Query|!- select decisioner, sum(mdcount + sdcount)::int as Count from
(
select case when decisioner is null then 'System' when decisioner='' then 'System' else decisioner end as decisioner,
sum(case when decisiontype ='Manual' then 1 else 0 end) as MDCount,
sum(case when decisiontype ='System' then 1 else 0 end) as SDCount,
RANK() OVER (PARTITION BY applicationnumber, producttype, productid ORDER BY applicationupdatetimestamp desc) as rnk
from f_application_decision where decision ='DECLINE' 
group by decisioner, applicationupdatetimestamp, applicationnumber, producttype, productid
) x where rnk=1
group by decisioner
order by decisioner -!|source_data|
