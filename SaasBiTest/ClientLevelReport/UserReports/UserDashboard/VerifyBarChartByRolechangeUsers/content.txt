!*> Compare source data with target data
!*< Setup
!|dbfit.PostgresTest                       |

|set option|bind symbols|false|

!*> Source Query
!|Store Query|!- select week, count(userid)::int as RoleChange_Cnt from 
(
select fuaa.userid, fuaa.eventname, fuaa.eventtimestamp, dd.weekdesc as week

where eventname='RoleChange' 
) x 
group by week
order by week -!|source_data|
